public boolean connectDB() {
	
	Properties prop = new Properties(); 
	prop.load("db.properties"); 
	
	username = prop.get("username"); 
	password = prop.get("password"); 
	
	Class.for("com.mysql.jdbc.Driver"); 
	Connection con = DriverManager.getConnection("jdbc:mysql://localhost:3306"prop.get("db") +, username, pass); 
	return conn!=null; 
	
}
